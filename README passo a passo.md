# Inicializar
git init

# Adicionando scripts
git add .

# Criando commit para mandar para nuvem
git commit -m "comentarios"

# Mnadando para nuvem
git push

# Vai pegar tudo que tem no main (principal) e joga lá ramificação dev
# e aqui ja entra na branch do dev
git checkout -b dev

# voltando pra branch do main
git switch main
